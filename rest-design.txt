// order api
/api/orders
	POST - Add order: List<Order> class type
	
/api/orders?List<Order>
	Get - get product price of each item in this order
	
/api/orders?customerID={id}&orderNum=111
	GET - find specific order

/api/orders?customerID={id}
	GET - get all order history of customerID {id}
	
/api/orders?mostrecent=true
	GET - mostrecent order/also able to get order number

/api/orders?customerID={id}&locationID=1
	GET - get location order history of customer {id}
